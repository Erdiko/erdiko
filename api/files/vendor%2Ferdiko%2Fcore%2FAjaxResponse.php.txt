namespace erdiko\core;
use Erdiko;

/**
 * AjaxResponse class
 */
class AjaxResponse extends Response 
{
    /**
     * Theme
     */
	protected $_theme;
    /**
     * Content
     */
	protected $_content = null;

    /**
     * Ajax render function
     *
     * @return string
     */
    public function render()
    {
        $responseData = array(
            "status" => 500,
            "body" => $this->_content,
            "errors" => array()
            );

        return json_encode($responseData);
    }

}
